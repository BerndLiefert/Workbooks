Schleifen:

For-Schleife
for(var i=0; i < myarray.lengt; i++){
console.log(myarray[i]);
}

While-Schleife

var i = 10;
while(i > 0){
console.log(i);
i--;
}

Do-While-Schleife

var i = 1;
var text = "";
do{
text  += i + "\n";
i++;
} while(i <= 10);
alert(text);


For-In-Schleife

var array = [1, 2, 3];
for (var item in array){
console.log(item);
//gibt den Index aus, also 0 1 2
}


var myarray = [1, 2, 3];

for(var step in myarray){
console.log(myarray[step]);
//gibt den Wert mit dem Index step aus, also 1 2 3
}

}
